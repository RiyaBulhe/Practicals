-- Half Adder VHDL code
library IEEE;
use IEEE.STD_LOGIC_1164.ALL;

entity HalfAdder is
    Port ( A : in  STD_LOGIC;
           B : in  STD_LOGIC;
           Sum : out  STD_LOGIC;
           Carry : out  STD_LOGIC);
end HalfAdder;

architecture Behavioral of HalfAdder is
begin
    -- Sum is XOR of A and B
    Sum <= A XOR B;
    
    -- Carry is AND of A and B
    Carry <= A AND B;
end Behavioral;
