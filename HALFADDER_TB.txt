-- Testbench for Half Adder
library IEEE;
use IEEE.STD_LOGIC_1164.ALL;

entity tb_HalfAdder is
end tb_HalfAdder;

architecture sim of tb_HalfAdder is
    -- Component declaration of the Half Adder
    component HalfAdder
        Port ( A : in  STD_LOGIC;
               B : in  STD_LOGIC;
               Sum : out  STD_LOGIC;
               Carry : out  STD_LOGIC);
    end component;

    -- Signals to connect to the HalfAdder
    signal A : STD_LOGIC := '0';
    signal B : STD_LOGIC := '0';
    signal Sum : STD_LOGIC;
    signal Carry : STD_LOGIC;
begin
    -- Instantiate the HalfAdder
    uut: HalfAdder port map (A => A, B => B, Sum => Sum, Carry => Carry);

    -- Test the Half Adder with different input combinations
    process
    begin
        -- Test case 1: A = 0, B = 0
        A <= '0'; B <= '0';
        wait for 10 ns;

        -- Test case 2: A = 0, B = 1
        A <= '0'; B <= '1';
        wait for 10 ns;

        -- Test case 3: A = 1, B = 0
        A <= '1'; B <= '0';
        wait for 10 ns;

        -- Test case 4: A = 1, B = 1
        A <= '1'; B <= '1';
        wait for 10 ns;

        -- End simulation
        wait;
    end process;
end sim;
